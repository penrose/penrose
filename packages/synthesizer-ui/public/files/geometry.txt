-- ~~~ TYPES ~~~
type Point
type Linelike
type Ray
type Line
type Segment

type Angle

type Rectangle

-- Angles

-- Polygons/Shapes

constructor MkRectangle : Point p * Point q * Point r * Point s -> Rectangle
constructor MkQuadrilateral : Point p * Point q * Point r * Point s -> Quadrilateral

function Diameter : Circle c * Point p * Point q -> Segment
